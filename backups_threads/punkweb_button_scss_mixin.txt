Mainly posting here to get these mixins/classes onto another computer, but incase anyone is interested in styling buttons similarly to this site here are the SCSS mixins.

[code]@mixin button($background, $color) {
  @include medium;
  appearance: none;
  background-color: $background;
  border: 1px solid mix(black, $background, 20%);
  border-radius: 2px;
  box-shadow: 1px 1px 1px rgba(0, 0, 0, .2);
  color: $color !important;
  cursor: pointer;
  display: inline-block;
  font-family: 'Open Sans';
  font-size: 13px;
  letter-spacing: 1px;
  outline: none;
  padding: .5rem .75rem;
  min-width: 100px;
  text-align: center;
  text-decoration: none !important;
  transition: all .25s ease-in-out;
  &:hover {
    box-shadow: 1px 1px 1px rgba(0, 0, 0, .2), 0px 0px 6px mix(white, $background, 10%);
    background-color: mix(black, $background, 10%);
  }
  &:disabled {
    background-color: $oc-gray-5;
    border: 1px solid $oc-gray-6;
    cursor: not-allowed;
    &:hover {
      box-shadow: 1px 1px 1px rgba(0, 0, 0, .2);
    }
  }
}

@mixin button-hollow($background, $color) {
  @include button($background, $color);
  background: transparent;
  border: 1px solid darken($background, 10%);
  color: $background !important;
  &:hover {
    background-color: darken($background, 5%);
    color: $color !important;
  }
  &:active {
    background-color: darken($background, 10%);
    color: $color !important;
  }
  &:disabled {
    background-color: transparent;
    border: 1px solid $oc-gray-5;
    color: $oc-gray-7 !important;
  }
}

.button {
  @include button($oc-gray-7, white);
  &-primary {
    @include button($primary, white);
  }
  &-green {
    @include button($oc-green-7, white);
  }
  &-red {
    @include button($oc-red-9, white);
  }
  &-orange {
    @include button($oc-orange-9, black);
  }
}

.button-hollow {
  @include button-hollow($oc-gray-7, white);
  &-primary {
    @include button-hollow($primary, white);
  }
  &-green {
    @include button-hollow($oc-green-7, white);
  }
  &-red {
    @include button-hollow($oc-red-9, white);
  }
  &-orange {
    @include button-hollow($oc-orange-7, white);
  }
}
[/code]
